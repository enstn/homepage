import { BrainVisualization } from './brain.js';
import { setupUIInteractions } from './ui.js';
import { devTools } from './utils.js';

// Make devTools available globally
window.devTools = devTools;

document.addEventListener('DOMContentLoaded', async () => {
    try {
        const brainViz = new BrainVisualization();
        
        setupUIInteractions();
        
        const loadingScreen = document.getElementById('loading-screen');
        setTimeout(() => {
            loadingScreen.style.opacity = '0';
            setTimeout(() => {
                loadingScreen.style.display = 'none';
            }, 500);
        }, 1000);
    } catch (error) {
        console.error('Error initializing brain:', error);
    }
});
